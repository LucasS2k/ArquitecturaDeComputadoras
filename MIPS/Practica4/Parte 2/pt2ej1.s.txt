.data
base: .word 5
exponente: .word 4
result: .word 0
.code

ld $a0, base($zero)
ld $a1, exponente($zero)  #$a parámetros de entrada
jal potencia #llamado a sub
sd $v0, result($zero) #se guarda el valor retornado por la sub
halt
potencia: daddi $v0, $zero, 1 #inicializa el resultado en 1
lazo: beqz $a1, terminar  #si el exponente es 0 termina
daddi $a1, $a1, -1 #decrementa exponente
dmul $v0, $v0, $a0 #multiplica v  por si mismo (v0 es el parámetro de salida)
j lazo
terminar: jr $ra #regresa al pp
