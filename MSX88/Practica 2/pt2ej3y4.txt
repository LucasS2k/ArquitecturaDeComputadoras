
PA EQU 30h 
PB EQU 31h
CA EQU 32h
CB EQU 33h
ORG 1000h 
letra db "A"
org 3000h
inipio: mov al, 11111101b
out ca, al
mov al, 0
ret

org 3100h
poll: in al, pa
and al, 1
jz poll
ret

org 3200h
flanco: in al, pa ; primero envio 0
and al, 11111101b
out pa, al
or al, 00000010b
out pa, al
ret

org 3300h
imprimir: push ax
call poll
pop ax
out pb, al
call flanco
ret

org 2000h
call inipio
mov al, letra
call imprimir
int 0
end
