PA equ 30h
PB equ 31h
CA equ 32h
CB equ 33h
        org 1000h
prendida db "primer llave prendida"
apagada db "primer llave apagada"
finstring db ?
        org 2000h
      

        mov al, 00000000b
        out CB, al
        mov al, 11111111b
        out CA, al
        in al, PA ;me traigo la iinfo
        and al, 10000000b ;mascara
        cmp al, 0 ;comparo con 0
        jz esta_apagada ;si es 0 esta apagada y salto
        mov bx, offset prendida ;caso contrario esta prendida
        mov al, offset apagada - offset prendida
        jmp fin ;salto para evitar entrar en la instrucciones de apagada
        esta_apagada: mov bx, offset apagada
        mov al, offset finstring - offset apagada
        fin: int 7 ;imprime lo que esta en BX con la cantidad de datos en al
