
pa equ 30h
pb equ 31h
ca equ 32h
cb equ 33h
org 1000h 
msj db ?
cant db 5
org 3000h
ini_pio:
mov al, 00000001b ;ponemos el 1 en el busy
out ca, al
mov al, 00000000b ; ponemos el 0 para que sea todo salida
out cb, al
ret

org 3200h
poll:
in al, pa
and al, 00000001b
jnz poll ;compara hasta no ser 0
ret
org 3300h
flanco:
in al, pa
and al, 11111101b ;forzamos
out pa, al
in al, pa
or al, 00000010b ;forzamos 1
out pa, al
ret

org 3400h
imprimirchar: int 6
dec cant
push ax ;guardamos el char
call poll
pop ax 
mov al, [bx] ;recuperamos char
out pb, al ;pasamos el char a pb
call flanco
inc bx ; avanzamos en el string
cmp cant, 0
jnz imprimirchar
ret

org 2000h
call ini_pio ;config
mov bx, offset msj
call imprimirchar ;sub
hlt
end
